// 定义一个arrayqueue类
class ArrayQueue {
    private int maxSize;// 数组最大的长度
    private int front;// 指向头部的前一个值
    private int real;// 指向尾部
    private int arr[];// 存放数据的

    // 定义构造
    public ArrayQueue(int maxArraySize) {
        maxSize = maxArraySize;
        front = -1;
        real = -1;
        arr = new int[maxSize];
    }
    // 添加数据
    public void addQueue(int n) {
        // 先判断是否满了
        if (isFull()) {
            throw new RuntimeException("队列已满，无法添加");
        }
        real++;
        arr[real] = n;
    }

    // 取数据
    public int getQueue() {
        // 先判断为空
        if (isEmpty()) {
            throw new RuntimeException("队列为空。。，清添加后在取数据");
        }
        front++;
        return arr[front];
    }

    // 查看队列内容
    public void showQueue() {
        // 判断不为空
        if (!isEmpty()) {
            for (int i = 0; i < maxSize; i++) {
                System.out.print(i + "\t");
            }
        }
    }
    // 查看头节点
    public int headQueue() {
        if (!isEmpty()) {
            return arr[++front];
        }
        throw new RuntimeException("队列为空，清添加后在进行次错操作");
    }
    // 判断是否满了
    public boolean isFull() {
        return real == maxSize - 1;
    }
    // 判断为空
    public boolean isEmpty() {
        return front == real;
    }
}

                default:
                    System.exit(0);
            }
        }
    }
}
class CircleQueue{
    int front;//指向第一个元素
    int real;//指向最后一个元素的后一个元素
    int maxSize;//队列长度
    int[] arr;
    
    public CircleQueue(int maxSize) {
        this.maxSize = maxSize;
        arr = new int[maxSize];
        front = 0;
        real = 0;
    }
    //判断队列是否满
    public boolean isFull() {
        return (real + 1) % maxSize == front;
    }
    //判断队列是否空
    public boolean isEmpty() {
        return real == front;
    }
    //往队列里面加值
    public void addQueue(int num) {
        if (isFull()) {
            throw new  RuntimeException("队列已满，无法再往里面添加元素了");
        }
        arr[real] = num;
        real = (real + 1) % maxSize;
    }
    //往队列里面取值
    public int getQueue() {
        if (isEmpty()) {
            throw new RuntimeException("队列是空的，请添加元素后在进行取值操作..");
        }
        int value = arr[front];
        front = (front + 1) % maxSize;
        return value;
    }
    //展示头部数据
    public int getHeadQueue() {
        if (isEmpty()) {
            System.out.println("队列为空，请添加元素后再查看..");
        }
        return arr[front];
    }
    //查看队列
    public void showQueue() {
        if (isEmpty()) {
            System.out.println("队列为空，请添加元素后再查看..");
        }
        for (int i=front;i<front + size();i++) {
            System.out.println("第" + (i%maxSize) +"个元素是" + arr[i%maxSize] );
        }
    }
    //统计有效数据
    public int size() {
        return (real + maxSize - front) % maxSize;
    }
}

    public HeroNode getHead() {
        return head;
    }

    // 往链表末尾里面添加元素
    public void add(HeroNode node) {
        // 找到末尾,定义一个辅助变量
        HeroNode temp = head;
        while (true) {
            if (temp.next == null) {
                break;
            }
            temp = temp.next;
        }
        temp.next = node;
    }

    // 按照编号插入
    public void addByOrder(HeroNode heroNode) {
        HeroNode temp = head;
        boolean flag = false;// 用来标识编号是否已经存在
        while (true) {
            if (temp.next == null) {// 已经到末尾了
                break;
            }
            if (temp.next.no > heroNode.no) {//找到了
                break;
            } else if (temp.next.no == heroNode.no){
                flag = true;
                break;
            }
            temp = temp.next;
        }
        //判断flag是否false
        if (flag) {
            System.out.println("编号已经存在");
        } else {
            heroNode.next = temp.next;
            temp.next = heroNode;
        }
    }

    // 遍历链表
    public void show() {
        // 定义一个辅助变量
        HeroNode temp = head.next;
        // 判断链表是否空
        if (temp == null) {
            System.out.println("链表空，");
            return;
        }
        while (true) {
            if (temp == null) {
                break;
            }
            System.out.println(temp);
            temp = temp.next;
        }
    }
}

    public static void reverseList(HeroNode head) {
        if (head.next == null || head.next.next == null) {
            //说明没有节点或者只有一个节点
            return ;
        }
        HeroNode cur = head.next;
        HeroNode next,prev = null;
        //HeroNode reverseHead = new HeroNode(0,"","");
        while(cur != null) {
            next = cur.next;//将现在节点的下一个节点保存一下
            cur.next = prev;//将现在结点的下一个节点指向前一个节点
            prev = cur;//保存现在的节点
            cur = next;//将辅助箭头后移，进行下一次循环
        }
        head.next = prev;
    }

            }

            temp.name = newHeroNode.name;

        }
    }
}

    // 构建一个环形单向链表
    public void add(int nums) {
        if (nums < 1) {
            System.out.println("nums的值不正确");
            return;
        }
        Boy cur = null;
        for (int i = 1; i <= nums; i++) {
            Boy boy = new Boy(i);
            if (i == 1) {
                first = boy;
                first.next = first;
                cur = first;
            } else {
                cur.next = boy;
                boy.next = first;
                cur = boy;
            }
        }
    }

    public void show() {
        if (first == null) {
            System.out.println("没有任何小孩");
            return;
        }
        Boy cur = first;
        while (true) {
            System.out.println(cur);
            if (cur.next == first) {
                break;
            }
            cur = cur.next;
        }
    }
    /**
     * @param startNo 表示从第几个小孩开始数
     * @param countNum 表示数几下
     * @param nums 表示最初有多少小孩在圈内
     */
    //根据用户的输入，计算出小孩出圈的顺序
    public void countBoy(int startNo,int countNum,int nums) {
        if (first == null || startNo < 1 || startNo > nums) {
            System.out.println("参数输入有误，请稍后再试");
            return;
        }
        Boy helper = first;//helper指向链表末尾
        while(true) {
            if (helper.next == first) {
                break;
            }
            helper = helper.next;
        }
        //先定位那个孩子开始数
        for (int i=1;i<startNo;i++) {
            first = first.next;
            helper = helper.next;
        }
        while(true) {
            if (helper == first) {
                break;
            }
          //开始数数
            for (int j = 1;j<countNum;j++) {//同时移动m-1次
                first = first.next;
                helper = helper.next;
            }
            System.out.printf("小孩%d出圈", first.no);
            first = first.next;
            helper.next = first;
        }
        System.out.printf("最后留在圈中的小孩编号%d \n",helper.no);
    }

}

public class ArrayStackDemo {

            switch (key) {
                case "show":
                    arrayStack.show();
                    break;
                case "exit":
                    sc.close();
                    loop = false;
                    break;
                case "push":
                    System.out.println("请输入要入栈的值");
                    int value = sc.nextInt();
                    arrayStack.push(value);
                    break;
                case "pop":

                    System.out.println("出栈的数据:"+result);
                    break;
            }
        }
        System.out.println("程序退出~");
    }
}
class ArrayStack{
    private int maxSize;
    private int[] stack;
    private int top = -1;
    public ArrayStack(int maxSize) {
        this.maxSize = maxSize;
        stack = new int[this.maxSize];
    }  
    //入栈
    public void push(int value) {
        if (isFull()) {
            System.out.println("栈满了");
            return;
        }
        top++;
        stack[top] = value;
    }
    //出栈
    public int pop() {
        if (isEmpty()) {
            throw new RuntimeException("栈空");
        }
        int value = stack[top];
        top--;
        return value;
    }
    //显示方法
    public void show() {
        if (isEmpty()) {
            throw new RuntimeException("栈空");
        }
        for (int i=top;i>=0;i--) {
            System.out.printf("stack[%d]=%d\r\n",i,stack[i]);
        }
    }
    //判断栈空
    public boolean isEmpty() {
        return top == -1;
    }  
    //判断栈满
    public boolean isFull() {
        return top == maxSize-1;
    } 
}

            for (int j=0;j<arr.length-1-i;j++){//arr.length-1表示每轮比较次数=arr.length-1-i,这个i表示前面选出来的最大值的次数
                if (arr[j]>arr[j+1]){
                    flag = true;
                    temp = arr[j];
                    arr[j] = arr[j+1];
                    arr[j+1] = temp;
                }
            }
            if (!flag){//在一趟排序中，一次都没有交换，说明不需要进行后面的操作，本省就是有序的
                break;
            } else {
                flag = false;//重置flag，进行下次判断
            }
        }
     return arr;
    }

public static void insertSort(int[] arr) {
        for (int i=1;i<arr.length;i++) {
            int indexVal = arr[i];
            int index = i-1;
            while(index >= 0 && indexVal<arr[index]) {
                arr[index+1] = arr[index];
                index--;
            }
            arr[index+1] = indexVal;
        }

7.2、二叉树

二叉树：树有很多种，每个节点最多只能有两个子节点的的一种形式称为二叉树。

满二叉树：如果该二叉树的所有叶子节点都在最后一层，并且节点总数=2^n-1，n为层数，则我们称为满二叉树。

完全二叉树：如果该二叉树的所有叶子节点都在最后一层或者倒数第二层，而且最后一层的叶子节点在左边连续，倒数第二层的叶子节点在右边连续，我们称为完全二叉树。

    //前序遍历
    public static void preOrderTraversal(TreeNode root){
        if (root == null){
            return;
        }
        System.out.print(root.val);
        if (root.left != null) {
            //递归进行
            preOrderTraversal(root.left);
        }
        if (root.right != null){
            //向右边递归前序遍历
            preOrderTraversal(root.right);
        }
    }
    //中序遍历
    public static void InOrderTraversal(TreeNode root){
        if (root == null){
            return;
        }
        if (root.left != null){
            InOrderTraversal(root.left);
        }
        System.out.print(root.val);
        if (root.right != null){
            InOrderTraversal(root.right);
        }
    }
    //后序遍历
    public static void postOrderTraversal(TreeNode root){
        if (root == null){
            return;
        }
        if (root.left != null){
            postOrderTraversal(root.left);
        }
        if (root.right != null){
            postOrderTraversal(root.right);
        }
        System.out.print(root.val);
    }
    //二叉树删除
    public static void delNode(TreeNode root,int result){
        if (root == null){
            return;
        }
        if (root.left != null && root.left.val == result){
            root.left = null;
            return;
        }
        if (root.right != null && root.right.val == result){
            root.right = null;
            return;
        }
        if (root.left != null){
            delNode(root.left,result);
        }
        if (root.right != null){
            delNode(root.right,result);
        }
    }
}

class Solution {
    public int strStr(String str, String dest) {
        char[] s1 = str.toCharArray();
        char[] s2 = dest.toCharArray();
        int i=0,j=0;
        while(i<s1.length && j<s2.length){
            if (s1[i] == s2[j]){
                i++;
                j++;
            } else {
                i = i - (j-1);
                j = 0;
            }
        }
        if (j == s2.length){
            return i - j;
        } else {
            return -1;
        }
    }
}
